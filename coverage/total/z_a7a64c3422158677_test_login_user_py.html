19def test_login_success(): 

20 """✅ 測試登入成功，應取得 token 並回傳成功碼""" 

21 cred = {"username": "user", "password": "pass"} 

22 headers = {"Content-Type": "application/json"} 

23 mock_response = Mock(status_code=200) 

24 mock_response.json.return_value = {"token": "abc123"} 

25
